  id: 'Ch4Scene1',
  durationInSeconds: 12,
  fps: 30,
  width: 1920,
  height: 1080,
};

  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

  const { fps, durationInFrames } = useVideoConfig();

    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',

    x: seededRandom(i * 7) * 1920,
    y: seededRandom(i * 13) * 1080,

    size: seededRandom(i * 11) * 3 + 1,

          transform: `rotate(${frame * 0.5}deg)`,

        {Array.from({ length: 20 }, (_, i) => (
          <div key={i} style={{
            position: 'absolute',
            width: '200%', height: 1,
            background: 'rgba(255,255,255,0.015)',

import { useCurrentFrame, useVideoConfig, interpolate, Easing, spring, AbsoluteFill } from 'remotion';

export const compositionConfig = {
  id: 'Ch4Scene3',
  durationInSeconds: 12,
  fps: 30,
  width: 1920,
  height: 1080,
};

const seededRandom = (seed: number): number => {
  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

const Ch4Scene3: React.FC = () => {
  const frame = useCurrentFrame();
  const { fps, durationInFrames } = useVideoConfig();

  const sp = (delay: number, config?: object) =>
    spring({ frame: Math.max(0, frame - delay), fps, config: { damping: 12, mass: 0.4, stiffness: 120, ...config } });

  const exitOp = interpolate(frame, [durationInFrames - 15, durationInFrames], [1, 0], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  // Pie chart animation — the deceptive portfolio
  const pieReveal = interpolate(frame, [15, 70], [0, 1], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
    easing: Easing.out(Easing.cubic),
  });

  // "60%" dangerous zone reveal
  const dangerReveal = sp(60);
  const labelFade = sp(75);

  const warningPulse = Math.sin(frame * 0.15) * 0.3 + 0.7;

  // SVG Donut chart helpers
  const cx = 300, cy = 300, r = 220, stroke = 80;
  const circumference = 2 * Math.PI * r;

  // 60% stocks (RED), 40% other (GREY)
  const stocksAngle = 0.6 * 360 * pieReveal;
  const otherAngle = 0.4 * 360 * pieReveal;

  const polarToCartesian = (cx: number, cy: number, r: number, angle: number) => {
    const rad = ((angle - 90) * Math.PI) / 180;
    return { x: cx + r * Math.cos(rad), y: cy + r * Math.sin(rad) };
  };

  const describeArc = (cx: number, cy: number, r: number, startAngle: number, endAngle: number) => {
