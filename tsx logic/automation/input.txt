export const compositionConfig = {
  id: 'C4S1_HebelStruktur',
  durationInSeconds: 9,
  fps: 30,
  width: 1920,
  height: 1080,
};

const seededRandom = (seed: number): number => {
  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

  const frame = useCurrentFrame();
  const { fps, durationInFrames } = useVideoConfig();

    spring({ frame: Math.max(0, frame - delay), fps, config: { damping: 12, mass: 0.4, stiffness: 120, ...config } });

  const exitOp = interpolate(frame, [durationInFrames - 15, durationInFrames], [1, 0], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',

  // VIRAL: Split screen tear effect
  const tearProgress = interpolate(frame, [0, 20], [0, 1], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
    easing: Easing.out(Easing.cubic),
  });

  // Left panel slides from left
  const leftX = interpolate(frame, [0, 25], [-960, 0], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
    easing: Easing.out(Easing.expo),
  });

  // Right panel slides from right
  const rightX = interpolate(frame, [0, 25], [960, 0], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
    easing: Easing.out(Easing.expo),
  });

  // EMPÖRUNG word smash
  const empScale = sp(5, { damping: 6, stiffness: 200 });
  const empOp = interpolate(frame, [5, 20], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });

  // STRUKTUR word entrance
  const strukturScale = sp(35, { damping: 10, stiffness: 150 });
  const strukturOp = interpolate(frame, [35, 50], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });

  // Arrow animation
  const arrowW = interpolate(frame, [55, 80], [0, 280], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
    easing: Easing.out(Easing.cubic),
  });

  // Screen shake on impact
  const shakeX = frame >= 5 && frame <= 12 ? Math.sin(frame * 8) * 6 : 0;
  const shakeY = frame >= 5 && frame <= 12 ? Math.cos(frame * 7) * 4 : 0;

  // Floating particles
  const particles = Array.from({ length: 20 }, (_, i) => ({
    x: seededRandom(i * 7) * 1920,
    y: seededRandom(i * 13) * 1080,
    size: seededRandom(i * 3) * 4 + 2,
    speed: seededRandom(i * 5) * 0.3 + 0.1,
    opacity: seededRandom(i * 11) * 0.4 + 0.1,
  }));

  // VS divider pulse
  const vsPulse = 1 + Math.sin(frame * 0.15) * 0.08;

  // Geometric shapes
  const shape1Rot = frame * 0.4;
  const shape2Rot = -frame * 0.25;
  const shape3Scale = 1 + Math.sin(frame * 0.08) * 0.05;

  return (
    <AbsoluteFill style={{ backgroundColor: '#1C1C1C', overflow: 'hidden', opacity: exitOp }}>

      {/* Dynamic split gradient background */}
      <AbsoluteFill style={{ transform: `translate(${shakeX}px, ${shakeY}px)` }}>

        {/* LEFT SIDE - RED (Empörung) */}
        <div style={{
          position: 'absolute',
          left: 0, top: 0,
          width: 960, height: 1080,
          background: 'linear-gradient(135deg, #3D0000 0%, #1a0000 60%, #1C1C1C 100%)',
          transform: `translateX(${leftX}px)`,
          clipPath: 'polygon(0 0, 100% 0, 88% 100%, 0 100%)',
        }} />

        {/* RIGHT SIDE - YELLOW (Struktur) */}
        <div style={{
          position: 'absolute',
          right: 0, top: 0,
          width: 960, height: 1080,
          background: 'linear-gradient(225deg, #2a2000 0%, #1a1400 60%, #1C1C1C 100%)',
          transform: `translateX(${rightX}px)`,
          clipPath: 'polygon(12% 0, 100% 0, 100% 100%, 0 100%)',
        }} />

        {/* CENTER GLOW LINE */}
        <div style={{
          position: 'absolute',
          left: '50%', top: 0,
          width: 4, height: 1080,
          background: 'linear-gradient(180deg, transparent, #e9d700, #ff0000, transparent)',
          transform: `translateX(-50%) scaleX(${vsPulse})`,
          boxShadow: '0 0 30px #e9d700, 0 0 60px rgba(233,215,0,0.3)',
          opacity: tearProgress,
        }} />

        {/* Large geometric shape layers */}
        <svg style={{ position: 'absolute', top: 0, left: 0, width: '100%', height: '100%', opacity: 0.05 }}>
          <rect
            x={-100} y={200}
            width={500} height={500}
            fill="none"
            stroke="#ff0000"
            strokeWidth={2}
            transform={`rotate(${shape1Rot}, 200, 450)`}
          />
          <rect
            x={1520} y={150}
            width={600} height={600}
            fill="none"
            stroke="#e9d700"
            strokeWidth={2}
            transform={`rotate(${shape2Rot}, 1820, 450)`}
          />
          <circle
            cx={960} cy={540}
            r={400}
            fill="none"
            stroke="#F5F5F5"
            strokeWidth={1}
            transform={`scale(${shape3Scale})`}
            style={{ transformOrigin: '960px 540px' }}
          />
        </svg>

        {/* Motion grid texture */}
        <svg style={{ position: 'absolute', top: 0, left: 0, width: '100%', height: '100%', opacity: 0.04 }}>
          <defs>
            <pattern id="c4s1grid" x="0" y="0" width="60" height="60" patternUnits="userSpaceOnUse">
              <path d="M 60 0 L 0 0 0 60" fill="none" stroke="#F5F5F5" strokeWidth="0.5" />
            </pattern>
          </defs>
          <rect width="1920" height="1080" fill="url(#c4s1grid)" />
        </svg>

        {/* Floating particles */}
        {particles.map((p, i) => (
          <div key={i} style={{
            position: 'absolute',
            left: (p.x + frame * p.speed * 20) % 1920,
            top: (p.y + Math.sin(frame * 0.04 + i) * 15),
            width: p.size, height: p.size,
            borderRadius: '50%',
            backgroundColor: i % 3 === 0 ? '#e9d700' : i % 3 === 1 ? '#ff0000' : '#F5F5F5',
            opacity: p.opacity,
          }} />
        ))}

        {/* LEFT: EMPÖRUNG */}
        <div style={{
          position: 'absolute',
          left: 80, top: '50%',
          transform: `translateY(-50%) translateX(${leftX}px) scale(${empScale})`,
          opacity: empOp,
        }}>
          <div style={{
            fontFamily: 'Anton, sans-serif',
            fontSize: 110,
            color: '#ff0000',
            letterSpacing: '-2px',
            lineHeight: 1,
            textShadow: '0 0 40px rgba(255,0,0,0.8), 0 0 80px rgba(255,0,0,0.4)',
          }}>
            EMPÖRUNG
          </div>
          <div style={{
            fontFamily: 'Open Sans, sans-serif',
            fontSize: 36,
            color: '#F5F5F5',
            opacity: 0.7,
            marginTop: 16,
            fontWeight: 300,
          }}>
            führt dich nirgendwo hin
          </div>

          {/* X mark */}
          <svg width={80} height={80} style={{ marginTop: 20, opacity: empOp }}>
            <line x1={10} y1={10} x2={70} y2={70} stroke="#ff0000" strokeWidth={6} strokeLinecap="round" />

        </div>

        <div style={{
          position: 'absolute',
          left: '50%', top: '50%',
          transform: `translate(-50%, -50%) scale(${vsPulse})`,
          fontFamily: 'Anton, sans-serif',
          fontSize: 72,
          color: '#F5F5F5',
          opacity: tearProgress * 0.9,
          textShadow: '0 0 20px rgba(255,255,255,0.5)',
          zIndex: 10,
        }}>
          VS
        </div>

        {/* RIGHT: STRUKTUR */}
        <div style={{
          position: 'absolute',
          right: 80, top: '50%',
          transform: `translateY(-50%) translateX(${rightX * -1}px) scale(${strukturScale})`,
          opacity: strukturOp,
          textAlign: 'right',
        }}>
          <div style={{
            fontFamily: 'Anton, sans-serif',
            fontSize: 110,
            color: '#e9d700',
            letterSpacing: '-2px',

            textShadow: '0 0 40px rgba(233,215,0,0.8), 0 0 80px rgba(233,215,0,0.4)',

            fontFamily: 'Open Sans, sans-serif',

          position: 'absolute',

          display: 'flex', alignItems: 'center', gap: 16,

    easing: Easing.out(Easing.cubic),
  });
  const word2Op = interpolate(frame, [18, 36], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });
  const word2Y = interpolate(frame, [18, 36], [30, 0], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
    easing: Easing.out(Easing.cubic),
  });
  const word3Scale = sp(45, { damping: 6, stiffness: 200 });
  const word3Op = interpolate(frame, [45, 65], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });

  // Connection line
  const lineW = interpolate(frame, [20, 40], [0, 200], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
    easing: Easing.out(Easing.cubic),
  });
  const lineOp = interpolate(frame, [20, 40], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });

  // Breathing ambient glow
  const glowScale = 1 + Math.sin(frame * 0.06) * 0.12;
  const glowOp = 0.15 + Math.sin(frame * 0.06) * 0.05;

  // Subscribe CTA
  const ctaOp = interpolate(frame, [85, 105], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });
  const ctaScale = sp(85, { damping: 10, stiffness: 120 });

  // Floating orbs
  const orbs = Array.from({ length: 12 }, (_, i) => ({
    x: seededRandom(i * 7) * 1920,
    y: seededRandom(i * 13) * 1080,
    r: seededRandom(i * 3) * 60 + 20,
    speed: seededRandom(i * 5) * 0.3 + 0.05,
    opacity: seededRandom(i * 11) * 0.08 + 0.02,
  }));

  // Star particles
  const stars = Array.from({ length: 40 }, (_, i) => ({
    x: seededRandom(i * 7) * 1920,
    y: seededRandom(i * 13) * 1080,
    size: seededRandom(i * 3) * 3 + 1,
    twinkle: seededRandom(i * 5) * Math.PI * 2,
    opacity: seededRandom(i * 11) * 0.5 + 0.1,
  }));

  return (
    <AbsoluteFill style={{ backgroundColor: '#1C1C1C', overflow: 'hidden', opacity: exitOp }}>

      {/* Deep space background */}
      <div style={{
        position: 'absolute', inset: 0,
        background: 'radial-gradient(ellipse at center, #0d0d1a 0%, #1C1C1C 100%)',
      }} />

      {/* Ambient glow */}
      <div style={{
        position: 'absolute',
        left: '50%', top: '50%',
        transform: `translate(-50%, -50%) scale(${glowScale})`,
        width: 800, height: 800,
        borderRadius: '50%',
        background: 'radial-gradient(circle, rgba(233,215,0,0.15) 0%, transparent 70%)',
        opacity: glowOp,
      }} />

      {/* Geometric shape - large diamond */}
      <svg style={{ position: 'absolute', top: 0, left: 0, width: '100%', height: '100%', opacity: 0.04 }}>
        <polygon points="960,80 1680,540 960,1000 240,540"
          fill="none" stroke="#e9d700" strokeWidth={1}
          transform={`rotate(${frame * 0.15}, 960, 540)`}
        />
        <polygon points="960,200 1560,540 960,880 360,540"
          fill="none" stroke="#e9d700" strokeWidth={0.5}
          transform={`rotate(${-frame * 0.1}, 960, 540)`}
        />
      </svg>

      {/* Star field */}
      {stars.map((star, i) => (
        <div key={i} style={{
          position: 'absolute',
          left: star.x, top: star.y,
          width: star.size, height: star.size,
          borderRadius: '50%',
          backgroundColor: i % 4 === 0 ? '#e9d700' : '#F5F5F5',
          opacity: star.opacity * (0.5 + Math.sin(frame * 0.1 + star.twinkle) * 0.5),
        }} />
      ))}

      {/* Floating orbs */}
      {orbs.map((orb, i) => (
        <div key={i} style={{
          position: 'absolute',
          left: (orb.x + Math.sin(frame * orb.speed * 0.05 + i) * 30),
          top: (orb.y + Math.cos(frame * orb.speed * 0.04 + i) * 20),
          width: orb.r * 2, height: orb.r * 2,
          borderRadius: '50%',
          background: i % 2 === 0
            ? 'radial-gradient(circle, rgba(233,215,0,0.3), transparent)'
            : 'radial-gradient(circle, rgba(255,255,255,0.15), transparent)',
          opacity: orb.opacity,
        }} />
      ))}

      {/* MAIN CONTENT - center */}
      <div style={{
        position: 'absolute',
        left: '50%', top: '50%',
        transform: 'translate(-50%, -50%)',
        textAlign: 'center',
        width: 1400,
      }}>

        {/* FINANZIELLE LUFT */}
        <div style={{
          opacity: word1Op,
          transform: `translateY(${word1Y}px)`,
          marginBottom: 10,
        }}>
          <div style={{
            fontFamily: 'Anton, sans-serif',
            fontSize: 120,
            color: '#e9d700',
            lineHeight: 1,
            letterSpacing: -2,
            textShadow: '0 0 60px rgba(233,215,0,0.5)',
          }}>
            FINANZIELLE LUFT
          </div>
        </div>

        {/* Connection line + IST AUCH */}
        <div style={{
          display: 'flex', alignItems: 'center', justifyContent: 'center',
          gap: 30, marginBottom: 10,
          opacity: lineOp,
        }}>
          <div style={{
            width: lineW, height: 2,
            background: 'linear-gradient(90deg, transparent, #e9d700)',
            borderRadius: 1,
          }} />
          <div style={{
            fontFamily: 'Open Sans, sans-serif',
            fontSize: 48,
            color: '#9E9E9E',
            fontWeight: 200,
            letterSpacing: 8,
          }}>
            IST AUCH
          </div>
          <div style={{
            width: lineW, height: 2,
            background: 'linear-gradient(90deg, #e9d700, transparent)',
            borderRadius: 1,
          }} />
        </div>

        {/* PSYCHISCHE LUFT */}
        <div style={{
          opacity: word2Op,
          transform: `translateY(${word2Y}px)`,
          marginBottom: 50,
        }}>
          <div style={{
            fontFamily: 'Anton, sans-serif',
            fontSize: 120,
            color: '#F5F5F5',
            lineHeight: 1,
            letterSpacing: -2,
          }}>
            PSYCHISCHE LUFT
          </div>
        </div>

        {/* UNBEZAHLBAR */}
        <div style={{
          transform: `scale(${word3Scale})`,
          opacity: word3Op,
        }}>
          <div style={{
            display: 'inline-block',
            background: 'linear-gradient(135deg, rgba(233,215,0,0.12), rgba(233,215,0,0.04))',
            border: '2px solid rgba(233,215,0,0.6)',
            borderRadius: 24,
            padding: '20px 60px',
            boxShadow: '0 0 40px rgba(233,215,0,0.3), inset 0 0 40px rgba(233,215,0,0.05)',
          }}>
            <div style={{
              fontFamily: 'Anton, sans-serif',
              fontSize: 90,
              color: '#e9d700',
              letterSpacing: 4,
              textShadow: '0 0 40px rgba(233,215,0,0.7)',
            }}>
              UNBEZAHLBAR
            </div>
          </div>
        </div>
      </div>

      {/* Stick figure - free, floating */}
      <div style={{
        position: 'absolute',
        bottom: 60, left: 80,
        transform: `translateY(${Math.sin(frame * 0.08) * -15}px) scale(${sp(20)})`,
        opacity: sp(20),
      }}>
        <svg width={90} height={170} viewBox="0 0 90 170">
          {/* Golden aura */}
          <circle cx={45} cy={30} r={35} fill="#e9d700"
            opacity={0.08 + Math.sin(frame * 0.08) * 0.04} />
          <circle cx={45} cy={30} r={50} fill="none" stroke="#e9d700" strokeWidth={1}
            opacity={0.08 + Math.sin(frame * 0.06) * 0.04} />
          <circle cx={45} cy={27} r={20} fill="none" stroke="#F5F5F5" strokeWidth={2.5}
            transform={`rotate(${Math.sin(frame * 0.04) * 2}, 45, 27)`} />
          <circle cx={36} cy={25} r={7} fill="none" stroke="#e9d700" strokeWidth={2} />
          <circle cx={54} cy={25} r={7} fill="none" stroke="#e9d700" strokeWidth={2} />
          <line x1={43} y1={25} x2={47} y2={25} stroke="#e9d700" strokeWidth={2} />
          {/* Serene smile */}
          <path d="M 36 35 Q 45 43 54 35" fill="none" stroke="#e9d700" strokeWidth={2} strokeLinecap="round" />
          <line x1={45} y1={47} x2={45} y2={105}
            stroke="#F5F5F5" strokeWidth={2.5}
            style={{ transform: `translateY(${Math.sin(frame * 0.05) * 2}px)` }} />
          {/* Arms open - free */}
          <line x1={45} y1={68} x2={8} y2={55} stroke="#F5F5F5" strokeWidth={2.5} strokeLinecap="round" />
          <line x1={45} y1={68} x2={82} y2={55} stroke="#F5F5F5" strokeWidth={2.5} strokeLinecap="round" />
          <line x1={45} y1={105} x2={24} y2={148} stroke="#F5F5F5" strokeWidth={2.5} strokeLinecap="round" />
          <line x1={45} y1={105} x2={66} y2={148} stroke="#F5F5F5" strokeWidth={2.5} strokeLinecap="round" />
        </svg>
      </div>

      {/* Bottom: Subscribe CTA */}
      <div style={{
        position: 'absolute',
        bottom: 45, right: 80,
        opacity: ctaOp,
        transform: `scale(${ctaScale})`,
        transformOrigin: 'right bottom',
      }}>
        <div style={{
          background: 'linear-gradient(90deg, rgba(233,215,0,0.12), rgba(233,215,0,0.06))',
          border: '1px solid rgba(233,215,0,0.5)',
          borderRadius: 14,
          padding: '16px 28px',
          display: 'flex', alignItems: 'center', gap: 16,
        }}>
          <div style={{
            width: 14, height: 14,
            borderRadius: '50%',
            backgroundColor: '#ff0000',
            boxShadow: '0 0 10px rgba(255,0,0,0.8)',
            animation: 'none',
            opacity: 0.5 + Math.sin(frame * 0.3) * 0.5,
          }} />
          <div style={{
            fontFamily: 'Inter, sans-serif',
            fontSize: 28,
            fontWeight: 700,
            color: '#e9d700',
            letterSpacing: 1,
          }}>
            KANALABO = Wiederholung = Vermögen
          </div>
        </div>
      </div>

      {/* Vignette */}
      <div style={{
        position: 'absolute', inset: 0,
        background: 'radial-gradient(ellipse at center, transparent 40%, rgba(0,0,0,0.7) 100%)',
        pointerEvents: 'none',
      }} />

      {/* Watermark */}
      <div style={{
        position: 'absolute', bottom: 30, right: 45,
        fontFamily: 'Inter, sans-serif', fontSize: 22, fontWeight: 600,
        color: '#9E9E9E', opacity: 0.3,
      }}>
        Kux legt an
      </div>

    </AbsoluteFill>
  );
};

export default C4S10_FinanzielleFreiheit;
