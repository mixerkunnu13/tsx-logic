import React from 'react';
import {
  useCurrentFrame,
  useVideoConfig,
  interpolate,
  spring,
  AbsoluteFill,
  Easing,
} from 'remotion';

export const compositionConfig = {
  id: 'Chapter4Scene1',
  durationInSeconds: 12,
  fps: 30,
  width: 1920,
  height: 1080,
};

const seededRandom = (seed: number): number => {
  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

const Chapter4Scene1: React.FC = () => {
  const frame = useCurrentFrame();
  const { fps, durationInFrames } = useVideoConfig();

  const sp = (delay: number, config?: object) =>
    spring({
      frame: Math.max(0, frame - delay),
      fps,
      config: { damping: 12, mass: 0.4, stiffness: 120, ...config },
    });

  const exitOp = interpolate(
    frame,
    [durationInFrames - 15, durationInFrames],
    [1, 0],
    { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' }
  );

  // Camera zoom in
  const cameraScale = interpolate(frame, [0, durationInFrames], [1, 1.06], {
    extrapolateRight: 'clamp',
  });

  // Gradient shift
  const gradAngle = interpolate(frame, [0, durationInFrames], [200, 240], {
    extrapolateRight: 'clamp',
  });

  // Chain links data
  const chainLinks = Array.from({ length: 7 }, (_, i) => ({
    x: 960 - 180 + i * 60,
    y: 540,
    delay: 10 + i * 4,
    seed: i + 1,
  }));

  // Breaking animation
  const breakFrame = 180;
  const breakProgress = interpolate(frame, [breakFrame, breakFrame + 30], [0, 1], {
    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',
  });

  return (
    <AbsoluteFill
      style={{
        backgroundColor: '#1C1C1C',
        overflow: 'hidden',
        opacity: exitOp,
      }}
    >
      {/* Dynamic gradient background */}
      <AbsoluteFill
        style={{
          background: `linear-gradient(${gradAngle}deg, #1C1C1C 0%, #2D1A1A 35%, #1C1C1C 70%, #1A1A2D 100%)`,
          transform: `scale(${cameraScale})`,
        }}
      />

      {/* Animated shape layers */}
      {Array.from({ length: 6 }, (_, i) => {
        const s = seededRandom(i + 40);
        const size = 120 + s * 300;
        const startX = s * 1920;
        const startY = seededRandom(i + 41) * 1080;
        const drift = Math.sin(frame * 0.008 + i * 2) * 40;
        const rot = frame * 0.15 * (i % 2 === 0 ? 1 : -1);
        const scaleBreath = 1 + Math.sin(frame * 0.02 + i) * 0.05;
        const shapes = ['circle', 'rect', 'polygon'];
        const shape = shapes[i % 3];
        return (
          <div
            key={`shape-${i}`}
            style={{
              position: 'absolute',
              left: startX + drift,
              top: startY + Math.cos(frame * 0.006 + i) * 30,
              width: size,
              height: size,
              transform: `rotate(${rot}deg) scale(${scaleBreath})`,
              opacity: 0.04 + (i % 3) * 0.015,
            }}
          >
            <svg width={size} height={size} viewBox={`0 0 ${size} ${size}`}>
              {shape === 'circle' && (
                <circle cx={size / 2} cy={size / 2} r={size / 2 - 2} fill="none" stroke="#ff0000" strokeWidth="1.5" />
              )}
              {shape === 'rect' && (
                <rect x="2" y="2" width={size - 4} height={size - 4} fill="none" stroke="#FFC107" strokeWidth="1.5" rx="8" />
              )}
              {shape === 'polygon' && (
                <polygon
                  points={`${size / 2},4 ${size - 4},${size - 4} 4,${size - 4}`}
                  fill="none"
                  stroke="#F5F5F5"
                  strokeWidth="1.5"
                />
              )}
            </svg>
          </div>
        );
      })}

      {/* Motion texture — diagonal lines */}
      <AbsoluteFill style={{ opacity: 0.03 }}>
        <svg width="1920" height="1080">
          {Array.from({ length: 30 }, (_, i) => (
            <line
              key={`line-${i}`}
              x1={i * 70 + Math.sin(frame * 0.01 + i) * 10}
              y1={0}
              x2={i * 70 - 200 + Math.sin(frame * 0.01 + i) * 10}
              y2={1080}
              stroke="#ff0000"
              strokeWidth="0.8"
            />
          ))}
        </svg>
      </AbsoluteFill>

      {/* Floating particles */}
      {Array.from({ length: 35 }, (_, i) => {
        const s = seededRandom(i + 100);
        const px = s * 1920;
        const py = seededRandom(i + 101) * 1080;
        const pSize = 2 + s * 4;
        const floatY = Math.sin(frame * 0.03 + i * 0.7) * 20;
        const floatX = Math.cos(frame * 0.02 + i * 0.5) * 15;
        const pOp = 0.15 + Math.sin(frame * 0.05 + i) * 0.1;
        return (
          <div
            key={`p-${i}`}
            style={{
              position: 'absolute',
              left: px + floatX,
              top: py + floatY,
              width: pSize,
              height: pSize,
              borderRadius: '50%',
              backgroundColor: i % 3 === 0 ? '#ff0000' : i % 3 === 1 ? '#FFC107' : '#F5F5F5',
              opacity: pOp,
            }}
          />
        );
      })}

      {/* Main content */}
      <AbsoluteFill
        style={{
          transform: `scale(${cameraScale})`,
          display: 'flex',
          alignItems: 'center',
          justifyContent: 'center',
        }}
      >
        {/* Large warning icon top */}
        <div
          style={{
            position: 'absolute',
            top: 120,
            left: '50%',
            transform: `translateX(-50%) scale(${sp(5)})`,
            opacity: sp(5),

              stroke="#ff0000"
              strokeWidth="1.5"

            display: 'flex',
            alignItems: 'center',
            justifyContent: 'center',
          }}
        >
          {chainLinks.map((link, i) => {
            const linkSp = sp(link.delay);
            const isMiddle = i === 3;
            const breakOffsetY = isMiddle
              ? interpolate(breakProgress, [0, 1], [0, 80])
              : 0;
            const breakRot = isMiddle
              ? interpolate(breakProgress, [0, 1], [0, 45])
              : 0;
            const breakOp = isMiddle
              ? interpolate(breakProgress, [0, 0.8, 1], [1, 1, 0.3])
              : 1;
            return (
              <div
                key={`chain-${i}`}
                style={{
                  transform: `translateY(${Math.sin(frame * 0.04 + i) * 3 + breakOffsetY}px) rotate(${breakRot}deg) scale(${linkSp})`,
                  opacity: linkSp * breakOp,
                  margin: '0 -4px',
                }}
              >
                <svg width="60" height="36" viewBox="0 0 60 36">
                  <rect
                    x="3"
                    y="3"
                    width="54"
                    height="30"
                    rx="15"
                    fill="none"
                    stroke={isMiddle ? '#ff0000' : '#FFC107'}
                    strokeWidth="3"
                  />
                </svg>
              </div>
            );
          })}
        </div>

        {/* Headline text */}
        <div
          style={{
            position: 'absolute',
            top: 420,
            textAlign: 'center',
            transform: `translateY(${interpolate(sp(20), [0, 1], [60, 0])}px)`,
            opacity: sp(20),
          }}
        >
          <div
            style={{
              fontFamily: 'Yantra One',
              fontSize: 110,
              color: '#F5F5F5',
              lineHeight: 1.15,
              textShadow: '0 4px 40px rgba(255,0,0,0.3)',

            बच्चों की{' '}

        </div>

            position: 'absolute',

            {/* Glasses */}
