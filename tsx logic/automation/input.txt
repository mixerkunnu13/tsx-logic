import React from 'react';
import { useCurrentFrame, useVideoConfig, interpolate, spring, AbsoluteFill } from 'remotion';

export const compositionConfig = {
  id: 'Ch4_Scene1',
  durationInSeconds: 15,
  fps: 30,
  width: 1920,
  height: 1080,
};

const seededRandom = (seed: number): number => {
  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

// Character component
const StickFigure: React.FC<{ frame: number }> = ({ frame }) => {
  const breath = Math.sin(frame * 0.05) * 2;
  const nod = Math.sin(frame * 0.04) * 3;
  return (
    <div style={{ position: 'absolute', bottom: 50, left: 100, transform: `translateY(${breath}px)` }}>
      <div style={{ position: 'absolute', top: -20, left: -20, width: 140, height: 140, background: 'radial-gradient(circle, #FFC107 0%, transparent 70%)', opacity: 0.4 }} />
      <svg width="100" height="200" viewBox="0 0 100 200">
        <g transform={`rotate(${nod}, 50, 40)`}>
          <circle cx="50" cy="40" r="20" fill="none" stroke="#F5F5F5" strokeWidth="6" />
          <circle cx="42" cy="38" r="6" fill="none" stroke="#FFC107" strokeWidth="3" />
          <circle cx="58" cy="38" r="6" fill="none" stroke="#FFC107" strokeWidth="3" />
          <line x1="48" y1="38" x2="52" y2="38" stroke="#FFC107" strokeWidth="2" />
        </g>
        <line x1="50" y1="60" x2="50" y2="130" stroke="#F5F5F5" strokeWidth="6" />
        <line x1="50" y1="80" x2="20" y2="110" stroke="#F5F5F5" strokeWidth="6" />
        <line x1="50" y1="80" x2="80" y2="110" stroke="#F5F5F5" strokeWidth="6" />
        <line x1="50" y1="130" x2="30" y2="190" stroke="#F5F5F5" strokeWidth="6" />
        <line x1="50" y1="130" x2="70" y2="190" stroke="#F5F5F5" strokeWidth="6" />
      </svg>
    </div>
  );
};

const Ch4_Scene1: React.FC = () => {
  // Calculate the current frame
  const frame = useCurrentFrame();
  // Extract video properties
  const { fps, durationInFrames } = useVideoConfig();

  // Spring animation for smooth entry
  const sp = (delay: number, config?: object) =>
    spring({ frame: Math.max(0, frame - delay), fps, config: { damping: 12, mass: 0.4, stiffness: 120, ...config } });

  const exitOp = interpolate(frame, [durationInFrames - 15, durationInFrames], [1, 0], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  const bgZoom = interpolate(frame, [0, durationInFrames], [1, 1.1]);
  const textY = interpolate(sp(15), [0, 1], [100, 0]);

  return (
    <AbsoluteFill style={{ backgroundColor: '#1C1C1C', overflow: 'hidden', opacity: exitOp }}>
      {/* Dynamic gradient background */}
      <AbsoluteFill style={{
        background: 'linear-gradient(135deg, #1C1C1C 0%, #2D2D2D 50%, #ff0000 150%)',
        transform: `scale(${bgZoom})`
      }} />

      {/* Motion texture (Grids) */}
      <AbsoluteFill style={{
        backgroundImage: 'linear-gradient(#9E9E9E 1px, transparent 1px), linear-gradient(90deg, #9E9E9E 1px, transparent 1px)',
        backgroundSize: '100px 100px',
        opacity: 0.03,
        transform: `translate(${frame * 0.5}px, ${frame * 0.5}px)`
      }} />

      {/* Animated shape layers */}
      <div style={{
        position: 'absolute', top: '10%', right: '10%', width: 600, height: 600,
        border: '4px solid #F5F5F5', opacity: 0.05,
        transform: `rotate(${frame * 0.1}deg) scale(${1 + Math.sin(frame * 0.02) * 0.1})`
      }} />

      {/* Floating particles */}
      {[...Array(10)].map((_, i) => (
        <div key={i} style={{
          position: 'absolute',
          left: `${seededRandom(i) * 100}%`,
          top: `${(seededRandom(i + 10) * 100) - (frame * 0.5)}%`,
          width: 8, height: 8, backgroundColor: '#FFC107', borderRadius: '50%',
          opacity: 0.4 + Math.sin(frame * 0.1 + i) * 0.3
        }} />
      ))}

      {/* Main Content */}
      <div style={{ position: 'absolute', width: '100%', top: '30%', textAlign: 'center', transform: `translateY(${textY}px)` }}>
        <h1 style={{ fontFamily: 'Yantra One', fontSize: 180, color: '#ff0000', margin: 0, textShadow: '0 10px 30px rgba(0,0,0,0.5)', opacity: sp(10) }}>LIFESTYLE</h1>
        <h2 style={{ fontFamily: 'Inter', fontSize: 120, color: '#F5F5F5', margin: '-20px 0 0 0', fontWeight: 900, opacity: sp(25) }}>INFLATION</h2>
        <p style={{ fontFamily: 'Noto Sans Devanagari', fontSize: 60, color: '#FFC107', marginTop: 30, opacity: sp(40) }}>
          सबसे छुपा हुआ खर्च
        </p>
      </div>

      <StickFigure frame={frame} />

      {/* Vignette */}
      <AbsoluteFill style={{ background: 'radial-gradient(circle, transparent 50%, rgba(0,0,0,0.7) 100%)' }} />

      {/* Watermark */}
      <div style={{ position: 'absolute', bottom: 30, right: 45, fontFamily: 'Inter', fontSize: 22, fontWeight: 600, color: '#9E9E9E', opacity: 0.3 }}>
        ProBuffet HQ
      </div>
    </AbsoluteFill>
  );
};

export default Ch4_Scene1;
