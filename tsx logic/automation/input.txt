import React from 'react';
import { useCurrentFrame, useVideoConfig, interpolate, Easing, spring, AbsoluteFill } from 'remotion';

export const compositionConfig = {
  id: 'Scene1_OverworkingTrap',
  durationInSeconds: 12,
  fps: 30,
  width: 1920,
  height: 1080,
};

const seededRandom = (seed: number): number => {
  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

const Scene1_OverworkingTrap: React.FC = () => {
  const frame = useCurrentFrame();
  const { fps, durationInFrames } = useVideoConfig();

  const sp = (delay: number, config?: object) =>
    spring({ frame: Math.max(0, frame - delay), fps, config: { damping: 12, mass: 0.4, stiffness: 120, ...config } });

  const exitOp = interpolate(frame, [durationInFrames - 15, durationInFrames], [1, 0], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  // Screen shake on impact
  const shakeX = frame > 20 && frame < 35 ? Math.sin(frame * 1.8) * 6 : 0;
  const shakeY = frame > 20 && frame < 35 ? Math.cos(frame * 2.1) * 4 : 0;

  // Trap number bounce
  const trapScale = sp(0, { damping: 8, stiffness: 200, mass: 0.3 });
  const titleReveal = sp(18);
  const subtitleReveal = sp(35);

  // Diagonal warning stripes animation
  const stripeOffset = (frame * 2) % 80;

  // Floating particles
  const particles = Array.from({ length: 18 }, (_, i) => ({
    x: seededRandom(i * 7) * 1920,
    y: seededRandom(i * 13) * 1080,
    size: seededRandom(i * 3) * 6 + 2,
    speed: seededRandom(i * 17) * 0.4 + 0.1,
    phase: seededRandom(i * 23) * Math.PI * 2,
  }));

  // Geometric shapes
  const shapes = [
    { x: 120, y: 180, size: 320, rotation: frame * 0.08, type: 'rect' },
    { x: 1700, y: 800, size: 200, rotation: -frame * 0.06, type: 'circle' },
    { x: 1500, y: 120, size: 260, rotation: frame * 0.04, type: 'triangle' },
    { x: 200, y: 900, size: 180, rotation: -frame * 0.1, type: 'rect' },
  ];

  return (
    <AbsoluteFill style={{
      backgroundColor: '#1C1C1C',
      overflow: 'hidden',
      opacity: exitOp,
      transform: `translate(${shakeX}px, ${shakeY}px)`,
    }}>
      {/* Gradient Background — Deep red warning gradient */}
      <AbsoluteFill style={{
        background: 'radial-gradient(ellipse at 30% 50%, #2a0a0a 0%, #1C1C1C 55%, #0d0d0d 100%)',
      }} />

      {/* Diagonal warning stripes — subtle */}
      <AbsoluteFill style={{ opacity: 0.04 }}>
        <svg width="1920" height="1080">
          {Array.from({ length: 30 }, (_, i) => (
            <line
              key={i}
              x1={i * 80 - 400 + stripeOffset}
              y1={0}
              x2={i * 80 - 400 + stripeOffset + 600}
              y2={1080}
              stroke="#ff0000"
              strokeWidth="40"
            />
          ))}
        </svg>
      </AbsoluteFill>

      {/* Shape Layers */}
      <AbsoluteFill style={{ opacity: 0 }}>
        {shapes.map((s, i) => (
          <div key={i} style={{
            position: 'absolute',
            left: s.x,
            top: s.y,
            width: s.size,
            height: s.size,
            transform: `rotate(${s.rotation}deg)`,
            opacity: 0.05 + seededRandom(i * 4) * 0.03,
          }}>
            {s.type === 'circle' && (
              <div style={{ width: '100%', height: '100%', borderRadius: '50%', border: '2px solid #ff0000' }} />
            )}
            {s.type === 'rect' && (
              <div style={{ width: '100%', height: '100%', border: '2px solid #e9d700' }} />
            )}
            {s.type === 'triangle' && (
              <svg width={s.size} height={s.size}>
                <polygon points={`${s.size / 2},0 ${s.size},${s.size} 0,${s.size}`} fill="none" stroke="#ff0000" strokeWidth="2" />
              </svg>
            )}
          </div>
        ))}
      </AbsoluteFill>

      {/* Motion Texture — Grid lines */}
      <AbsoluteFill>
        <svg width="1920" height="1080" style={{ opacity: 0.04 }}>
          {Array.from({ length: 24 }, (_, i) => (
            <line key={`v${i}`} x1={i * 80} y1={0} x2={i * 80} y2={1080} stroke="#9E9E9E" strokeWidth="1" />
          ))}
          {Array.from({ length: 14 }, (_, i) => (
            <line key={`h${i}`} x1={0} y1={i * 80} x2={1920} y2={i * 80} stroke="#9E9E9E" strokeWidth="1" />
          ))}
        </svg>
      </AbsoluteFill>

      {/* Floating Particles */}
      <AbsoluteFill>
        {particles.map((p, i) => (
          <div key={i} style={{
            position: 'absolute',
            left: p.x + Math.sin(frame * p.speed + p.phase) * 15,
            top: p.y + Math.cos(frame * p.speed * 0.7 + p.phase) * 10,
            width: p.size,
            height: p.size,
            borderRadius: '50%',
            backgroundColor: i % 3 === 0 ? '#ff0000' : i % 3 === 1 ? '#e9d700' : '#F5F5F5',

          letterSpacing: '-10px',
        }}>05</div>
      </div>

        position: 'absolute',
        left: 140,

        transform: `translateX(${interpolate(sp(18), [0, 1], [-80, 0])}px)`,
        opacity: titleReveal,
      }}>
        <div style={{
          fontFamily: 'Yantra One',
          fontSize: 130,
          fontWeight: 900,
          color: '#F5F5F5',
          lineHeight: 1.05,
          letterSpacing: '-2px',
        }}>
          पाँचवाँ जाल
        </div>
        <div style={{
          fontFamily: 'Inter, Poppins',
          fontSize: 72,
          fontWeight: 800,
          color: '#ff0000',
          letterSpacing: '4px',
          textTransform: 'uppercase',
          marginTop: 8,
        }}>
          OVERWORKING
        </div>
      </div>

      {/* Subtitle */}
      <div style={{
        position: 'absolute',
        left: 140,
        top: 560,
        opacity: subtitleReveal,
        transform: `translateY(${interpolate(sp(35), [0, 1], [30, 0])}px)`,
